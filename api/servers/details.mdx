---
title: 'Get Server Details'
description: 'Retrieve detailed information about a specific server'
api: 'GET /api/client/servers/:server'
---

Retrieves comprehensive details about a specific server, including resource limits, allocations, permissions, and current status.

## Authentication

Requires a valid session token and the `server.view` permission on the specified server.

## Path Parameters

<ParamField path="server" type="string" required>
  Server identifier (short identifier or UUID)
</ParamField>

## Response

<ResponseField name="data" type="object">
  Server details object
  <Expandable title="PanelServerDetails">
    <ResponseField name="id" type="string" required>
      Internal database ID of the server
    </ResponseField>
    <ResponseField name="uuid" type="string" required>
      Unique identifier for the server
    </ResponseField>
    <ResponseField name="identifier" type="string" required>
      Short identifier used in URLs
    </ResponseField>
    <ResponseField name="name" type="string" required>
      Display name of the server
    </ResponseField>
    <ResponseField name="description" type="string | null">
      Server description
    </ResponseField>
    <ResponseField name="status" type="string | null">
      Current server status: `offline`, `starting`, `running`, `stopping`, or `stopped`
    </ResponseField>
    <ResponseField name="suspended" type="boolean" required>
      Whether the server is currently suspended
    </ResponseField>
    <ResponseField name="node" type="object" required>
      Information about the Wings node hosting this server
      <Expandable title="Node">
        <ResponseField name="id" type="string | null">
          Node ID
        </ResponseField>
        <ResponseField name="name" type="string | null">
          Node display name
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="limits" type="object" required>
      Server resource limits
      <Expandable title="Limits">
        <ResponseField name="memory" type="number | null">
          Memory limit in megabytes
        </ResponseField>
        <ResponseField name="disk" type="number | null">
          Disk limit in megabytes
        </ResponseField>
        <ResponseField name="cpu" type="number | null">
          CPU limit in percentage (100 = 1 core)
        </ResponseField>
        <ResponseField name="swap" type="number | null">
          Swap memory limit (reserved for future use)
        </ResponseField>
        <ResponseField name="io" type="number | null">
          IO weight/priority
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="createdAt" type="string" required>
      ISO 8601 timestamp of when the server was created
    </ResponseField>
    <ResponseField name="allocations" type="object" required>
      Network allocations for the server
      <Expandable title="Allocations">
        <ResponseField name="primary" type="object | null">
          Primary network allocation
          <Expandable title="Allocation">
            <ResponseField name="ip" type="string">
              IP address
            </ResponseField>
            <ResponseField name="port" type="number">
              Port number
            </ResponseField>
            <ResponseField name="description" type="string">
              Allocation description or notes
            </ResponseField>
          </Expandable>
        </ResponseField>
        <ResponseField name="additional" type="array">
          Array of additional network allocations with the same structure as primary
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="owner" type="object" required>
      Information about the server owner
      <Expandable title="Owner">
        <ResponseField name="id" type="string | null">
          Owner user ID
        </ResponseField>
        <ResponseField name="username" type="string | null">
          Owner username
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="permissions" type="array">
      Array of permission strings the current user has on this server
    </ResponseField>
  </Expandable>
</ResponseField>

## Example Request

```bash
curl -X GET "https://panel.example.com/api/client/servers/my-server" \
  -H "Cookie: session=your-session-token"
```

## Error Responses

<ResponseField name="400" type="Bad Request">
  Server identifier is missing or invalid
</ResponseField>

<ResponseField name="403" type="Forbidden">
  User does not have the `server.view` permission
</ResponseField>

<ResponseField name="404" type="Not Found">
  Server not found or user does not have access
</ResponseField>
